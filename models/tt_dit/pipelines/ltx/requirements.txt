# LTX (audio/video) pipeline — supplemental requirements.
#
# These are intentionally kept OUT of the shared tt_metal/python_env/requirements-dev.txt
# so that bumping deps for LTX does not affect everyone else's environment.
#
# Install ON TOP OF the normal dev env, AFTER running ./create_venv.sh:
#
#     source python_env/bin/activate
#     uv pip install --prerelease=allow -r models/tt_dit/pipelines/ltx/requirements.txt
#
# Why --prerelease=allow:
#   diffusers==0.38.0 requires safetensors>=0.8.0rc0, and there is currently no
#   *stable* safetensors >=0.8.0 on PyPI (latest stable is 0.7.0). The flag lets
#   uv/pip select the 0.8.0rc pre-release. Drop the flag once safetensors 0.8.0
#   ships a stable release.

# Newer diffusers required by the LTX pipeline (shared env pins 0.35.1).
diffusers==0.38.0

# PyAV — used by models/tt_dit/utils/video.py to mux generated frames + audio
# into the output .mp4.
av==17.0.1
