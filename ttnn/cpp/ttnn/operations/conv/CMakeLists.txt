    ttnn_op_conv

        VERIFY_INTERFACE_HEADER_SETS

    ttnn_op_conv
    PRIVATE
        conv2d/conv2d.cpp
        conv2d/conv2d_utils.cpp
        conv2d/conv2d_op_program_factory_common.cpp
        conv2d/device/conv2d_op.cpp

    ttnn_op_conv

        BASE_DIRS ${CMAKE_CURRENT_SOURCE_DIR}
        FILES ${kernels}

    ttnn_op_conv

    TARGETS
        ttnn_op_conv
    FILE_SET
    kernels
