# Whisper 2CQ Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add dual command queue (2CQ) support to Whisper's traced decoder, overlapping host-to-device transfers (Q1) with computation (Q0).

**Architecture:** Replace the existing single-CQ trace capture/execute methods in `WhisperGenerator` with 2CQ versions following the sentence_bert pattern. Q1 handles host-to-device token ID transfers while Q0 runs embedding + position add + decoder trace. The trace boundary (decoder only) is unchanged.

**Tech Stack:** ttnn (Tenstorrent neural network library), Python, pytest

**Spec:** `docs/superpowers/specs/2026-03-14-whisper-2cq-design.md`

**Reference implementations:**
- Sentence_bert 2CQ: `models/demos/sentence_bert/runner/performant_runner.py`
- Current Whisper trace: `models/demos/audio/whisper/tt/whisper_generator.py`

**Important note for implementers:** Line numbers referenced in this plan are from the original source files. After each task modifies a file, subsequent line numbers shift. Always locate code by searching for function/method names and surrounding context, not by line number alone.

```python
    def _capture_decoder_trace(self, trace_key, decode_pos):
        """
        Capture decoder trace with 2CQ event protocol.

        Three-phase pattern following sentence_bert:
        Phase 1 (JIT): Event-synchronized run for JIT compilation
        Phase 2 (Optimized): Event-synchronized run for optimization
        Phase 3 (Capture): Actual trace capture with buffer address validation

        Args:
            trace_key: Batch size per device key for trace lookup
            decode_pos: Host-side integer decode position for position embedding
        """
        if self.trace_id_decoder[trace_key] is not None:
            return  # Already captured

        # Create decoder function that will be traced
        # cross_attn_cache_valid=True because cache was just populated in iteration 0
        def traced_decoder_fn(trace_key, hidden_states):
            return ttnn_optimized_functional_whisper.decoder(
                self.config,
                hidden_states,
                decoder_attention_mask=None,
                encoder_hidden_states=self.encoder_hidden_states_per_size[trace_key],
                kv_cache=self.kv_cache_per_batch_size[trace_key],
                cross_attn_cache=self.cross_attn_cache_per_batch_size[trace_key],
                cross_attn_cache_valid=True,
                current_decode_pos=self.current_decode_pos_per_size[trace_key],
                parameters=self.parameters.decoder,
            )

        # Helper: run event-synchronized input transfer + embedding + position add
        # (intentionally duplicates logic from preprocess_decoder_inputs to avoid
        # modifying the functional module; see spec Non-Goals)
        def _run_2cq_preprocessing(token_id_host, decode_pos):
            ttnn.wait_for_event(1, self.op_event)
            ttnn.copy_host_to_device_tensor(token_id_host, self.token_id_device[trace_key], 1)
            self.write_event = ttnn.record_event(self.mesh_device, 1)
            ttnn.wait_for_event(0, self.write_event)
            # Embedding lookup on Q0
            inputs_embeds = ttnn.embedding(
                self.token_id_device[trace_key],
                self.parameters.decoder.embed_tokens.weight,
                layout=ttnn.TILE_LAYOUT,
                memory_config=ttnn.DRAM_MEMORY_CONFIG,
            )
            # Position add on Q0 (host-integer indexed, outside trace)
            positions = self.parameters.decoder.embed_positions.weight[decode_pos : decode_pos + 1]
            positions = ttnn.to_layout(positions, ttnn.TILE_LAYOUT)
            decoder_hidden_states = inputs_embeds + positions
            return decoder_hidden_states

        # Prepare host staging tensor from sample input
        # We need a host token ID tensor; create a dummy one for trace capture phases
        # (torch is already imported at module level)
        dummy_token_ids = torch.zeros((trace_key, 1), dtype=torch.long)
        token_id_host = ttnn.from_torch(
            dummy_token_ids, dtype=ttnn.uint32, layout=ttnn.ROW_MAJOR_LAYOUT,
            mesh_mapper=self.input_mesh_mapper,
        )
        self.token_id_host[trace_key] = token_id_host

        # Initialize op_event
        self.op_event = ttnn.record_event(self.mesh_device, 0)

        # === Phase 1: JIT run ===
        decoder_hidden_states = _run_2cq_preprocessing(token_id_host, decode_pos)
        l1_input = ttnn.to_memory_config(decoder_hidden_states, ttnn.L1_MEMORY_CONFIG)
        self.op_event = ttnn.record_event(self.mesh_device, 0)
        compile_output = traced_decoder_fn(trace_key, l1_input)
        ttnn.deallocate(compile_output, force=True)
        ttnn.deallocate(l1_input)
        logger.info("Decoder trace 2CQ Phase 1 (JIT) complete")

        # === Phase 2: Optimized run ===
        decoder_hidden_states = _run_2cq_preprocessing(token_id_host, decode_pos)
        l1_input = ttnn.to_memory_config(decoder_hidden_states, ttnn.L1_MEMORY_CONFIG)
        self.op_event = ttnn.record_event(self.mesh_device, 0)
        optimized_output = traced_decoder_fn(trace_key, l1_input)
        ttnn.deallocate(optimized_output, force=True)
        ttnn.deallocate(l1_input)
        logger.info("Decoder trace 2CQ Phase 2 (optimized) complete")

        # === Phase 3: Trace capture ===
        decoder_hidden_states = _run_2cq_preprocessing(token_id_host, decode_pos)
        self.trace_input_decoder[trace_key] = ttnn.to_memory_config(
            decoder_hidden_states, ttnn.L1_MEMORY_CONFIG
        )
        ttnn.deallocate(decoder_hidden_states)  # Free DRAM intermediate
        self.op_event = ttnn.record_event(self.mesh_device, 0)

        # Record buffer address and spec before capture
        trace_input_addr = self.trace_input_decoder[trace_key].buffer_address()
        trace_input_spec = self.trace_input_decoder[trace_key].spec

        # Capture trace (decoder only)
        self.trace_id_decoder[trace_key] = ttnn.begin_trace_capture(self.mesh_device, cq_id=0)
        self.trace_output_decoder[trace_key] = traced_decoder_fn(
            trace_key, self.trace_input_decoder[trace_key]
        )

        # Allocate tensor with same spec to reclaim buffer address for next iteration
        self.trace_input_decoder[trace_key] = ttnn.allocate_tensor_on_device(
            trace_input_spec, self.mesh_device
        )

        ttnn.end_trace_capture(self.mesh_device, self.trace_id_decoder[trace_key], cq_id=0)
        ttnn.synchronize_device(self.mesh_device)

        # Validate buffer address stability
        assert trace_input_addr == self.trace_input_decoder[trace_key].buffer_address(), (
            f"Trace input buffer address changed: {trace_input_addr} -> "
            f"{self.trace_input_decoder[trace_key].buffer_address()}"
        )

        logger.info(
            f"Persistent decoder trace 2CQ capture complete for batch size per device {trace_key}"
        )
```

```python
    def _execute_decoder_trace(self, trace_key, token_id_host, decode_pos):
        """
        Execute the captured decoder trace with 2CQ event protocol.

        Q1 transfers token ID host->device while Q0 may still be finishing
        previous trace. Then Q0 runs embedding + position add + trace.

        Args:
            trace_key: Batch size per device key for trace lookup
            token_id_host: Host tensor with token IDs (uint32, shape [batch_size, 1])
            decode_pos: Host-side integer decode position for position embedding

        Returns:
            Decoder output tensor
        """
        if self.trace_id_decoder[trace_key] is None:
            raise RuntimeError("Decoder trace not captured. Call _capture_decoder_trace first.")

        # Q1: transfer token ID while Q0 may still be executing previous trace
        ttnn.wait_for_event(1, self.op_event)
        ttnn.copy_host_to_device_tensor(token_id_host, self.token_id_device[trace_key], 1)
        self.write_event = ttnn.record_event(self.mesh_device, 1)

        # Q0: wait for Q1 transfer, then run embedding + position add
        ttnn.wait_for_event(0, self.write_event)

        # Embedding lookup on Q0
        inputs_embeds = ttnn.embedding(
            self.token_id_device[trace_key],
            self.parameters.decoder.embed_tokens.weight,
            layout=ttnn.TILE_LAYOUT,
            memory_config=ttnn.DRAM_MEMORY_CONFIG,
        )

        # Position add on Q0 (host-integer indexed, outside trace)
        positions = self.parameters.decoder.embed_positions.weight[decode_pos : decode_pos + 1]
        positions = ttnn.to_layout(positions, ttnn.TILE_LAYOUT)
        decoder_hidden_states = inputs_embeds + positions

        # Copy to trace input L1 tensor at stable buffer address
        self.trace_input_decoder[trace_key] = ttnn.to_memory_config(
            decoder_hidden_states,
            ttnn.L1_MEMORY_CONFIG,
            output_tensor=self.trace_input_decoder[trace_key],
        )

        # Record event BEFORE trace execution (this is what enables overlap:
        # next iteration's Q1 transfer can start as soon as this event fires,
        # while Q0's trace is still running)
        self.op_event = ttnn.record_event(self.mesh_device, 0)

        # Execute trace (non-blocking to allow Q1 overlap on next iteration)
        ttnn.execute_trace(self.mesh_device, self.trace_id_decoder[trace_key], cq_id=0, blocking=False)

        # Sync before output read — required because to_torch reads the output tensor
        ttnn.synchronize_device(self.mesh_device)

        return self.trace_output_decoder[trace_key]
```
