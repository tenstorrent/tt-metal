<objective>
Rename 4 raw-size non-scatter function families in `mesh/api.h` to `_single_packet` and add auto-packetizing wrappers.

Purpose: mesh/api.h has the same families as linear but with mesh-specific routing (dst_dev_id, dst_mesh_id, MeshMcastRange). Key difference from linear: mesh raw functions have a `bool SetRoute = true` template parameter. Wrappers call `fabric_set_unicast_route` / `fabric_set_mcast_route` once before the loop, then use `SetRoute=false` on every _single_packet call to avoid redundant route re-initialization. For conn mgr variants, routes are set per-slot via a dedicated setup pass (for_each_header calling fabric_set_unicast_route) BEFORE the chunk loop; subsequent _single_packet calls inside the loop use `SetRoute=false`.

Output: Modified `tt_metal/fabric/hw/inc/mesh/api.h` with 8 renames + 8 auto-packetizing wrappers.
</objective>

Template wrapper:
```cpp
template <typename FabricSenderType>
FORCE_INLINE void fabric_unicast_noc_unicast_write(
    tt_l1_ptr FabricSenderType* client_interface,
    volatile PACKET_HEADER_TYPE* packet_header,
    uint8_t dst_dev_id,
    uint16_t dst_mesh_id,
    uint32_t src_addr,
    uint32_t size,
    tt::tt_fabric::NocUnicastCommandHeader noc_unicast_command_header) {
    // Set route once before loop
    fabric_set_unicast_route(packet_header, dst_dev_id, dst_mesh_id);

    uint32_t remaining = size;
    uint32_t current_src = src_addr;
    uint64_t current_noc_addr = noc_unicast_command_header.noc_address;

    while (remaining > FABRIC_MAX_PACKET_SIZE) {
        noc_async_writes_flushed();
        fabric_unicast_noc_unicast_write_single_packet<FabricSenderType, false>(
            client_interface, packet_header, dst_dev_id, dst_mesh_id, current_src,
            FABRIC_MAX_PACKET_SIZE, tt::tt_fabric::NocUnicastCommandHeader{current_noc_addr});
        current_src += FABRIC_MAX_PACKET_SIZE;
        current_noc_addr += FABRIC_MAX_PACKET_SIZE;
        remaining -= FABRIC_MAX_PACKET_SIZE;
    }
    noc_async_writes_flushed();
    fabric_unicast_noc_unicast_write_single_packet<FabricSenderType, false>(
        client_interface, packet_header, dst_dev_id, dst_mesh_id, current_src, remaining,
        tt::tt_fabric::NocUnicastCommandHeader{current_noc_addr});
}
```

Conn mgr wrapper (breadth-first, SetRoute=false per Pattern 5):
```cpp
FORCE_INLINE void fabric_unicast_noc_unicast_write(
    tt::tt_fabric::RoutingPlaneConnectionManager& connection_manager,
    uint8_t route_id,
    uint32_t src_addr,
    uint32_t size,
    tt::tt_fabric::NocUnicastCommandHeader noc_unicast_command_header) {
    uint32_t remaining = size;
    uint32_t current_src = src_addr;
    uint64_t current_noc_addr = noc_unicast_command_header.noc_address;

    // Route-setup pass: set route for each slot ONCE before the chunk loop
    PacketHeaderPool::for_each_header(route_id, [&](volatile PACKET_HEADER_TYPE* packet_header, uint8_t i) {
        auto& slot = connection_manager.get(i);
        fabric_set_unicast_route(packet_header, slot.dst_dev_id, slot.dst_mesh_id);
    });

    // Chunk loop: use SetRoute=false — route was already set above
    while (remaining > FABRIC_MAX_PACKET_SIZE) {
        noc_async_writes_flushed();
        PacketHeaderPool::for_each_header(route_id, [&](volatile PACKET_HEADER_TYPE* packet_header, uint8_t i) {
            auto& slot = connection_manager.get(i);
            fabric_unicast_noc_unicast_write_single_packet<decltype(slot.sender), false>(
                &slot.sender, packet_header, slot.dst_dev_id, slot.dst_mesh_id, current_src,
                FABRIC_MAX_PACKET_SIZE, tt::tt_fabric::NocUnicastCommandHeader{current_noc_addr});
        });
        current_src += FABRIC_MAX_PACKET_SIZE;
        current_noc_addr += FABRIC_MAX_PACKET_SIZE;
        remaining -= FABRIC_MAX_PACKET_SIZE;
    }
    // Final chunk (SetRoute=false)
    noc_async_writes_flushed();
    PacketHeaderPool::for_each_header(route_id, [&](volatile PACKET_HEADER_TYPE* packet_header, uint8_t i) {
        auto& slot = connection_manager.get(i);
        fabric_unicast_noc_unicast_write_single_packet<decltype(slot.sender), false>(
            &slot.sender, packet_header, slot.dst_dev_id, slot.dst_mesh_id, current_src, remaining,
            tt::tt_fabric::NocUnicastCommandHeader{current_noc_addr});
    });
}
```

NOTE: The route-setup `for_each_header` pass appears BEFORE the while loop, not inside it. Every call to `_single_packet` inside the chunk loop uses `<..., false>` — route is NOT re-set per chunk. This is Pattern 5 from RESEARCH.md.
